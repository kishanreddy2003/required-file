============================================

STEP-1:
======
<dependency>
			<groupId>org.springframework.cloud</groupId>
			<artifactId>spring-cloud-starter-circuitbreaker-reactor-resilience4j</artifactId>
		</dependency>


		<dependency>
			<groupId>org.springframework.boot</groupId>
			<artifactId>spring-boot-starter-actuator</artifactId>
		</dependency>



STEP-1:Add application.property
--------------------------------
spring.application.name=api-gateway

server.port=5555

management.endpoints.web.exposure.include=health,metrics,circuitbreakers
management.endpoint.health.show-details=always

# ===============================
# PROPERTY SERVICE ROUTE
# ===============================
spring.cloud.gateway.server.webflux.routes[0].id=property-service-api
spring.cloud.gateway.server.webflux.routes[0].uri=lb://PROPERTY-SERVICE
spring.cloud.gateway.server.webflux.routes[0].predicates[0]=Path=/property/**
spring.cloud.gateway.server.webflux.routes[0].filters[0]=RewritePath=/property/(?<segment>.*), /${segment}
# Filter 2 → Circuit Breaker
spring.cloud.gateway.server.webflux.routes[0].filters[1].name=CircuitBreaker
spring.cloud.gateway.server.webflux.routes[0].filters[1].args.name=propertyCircuitBreaker
spring.cloud.gateway.server.webflux.routes[0].filters[1].args.fallbackUri=forward:/fallback/property
# ===============================
# RESILIENCE4J CIRCUIT BREAKER
# ===============================
resilience4j.circuitbreaker.instances.propertyCircuitBreaker.registerHealthIndicator=true
resilience4j.circuitbreaker.instances.propertyCircuitBreaker.slidingWindowType=COUNT_BASED
resilience4j.circuitbreaker.instances.propertyCircuitBreaker.slidingWindowSize=20
resilience4j.circuitbreaker.instances.propertyCircuitBreaker.minimumNumberOfCalls=10
resilience4j.circuitbreaker.instances.propertyCircuitBreaker.failureRateThreshold=50
resilience4j.circuitbreaker.instances.propertyCircuitBreaker.waitDurationInOpenState=20s
resilience4j.circuitbreaker.instances.propertyCircuitBreaker.permittedNumberOfCallsInHalfOpenState=5
resilience4j.circuitbreaker.instances.propertyCircuitBreaker.automaticTransitionFromOpenToHalfOpenEnabled=true
resilience4j.circuitbreaker.instances.propertyCircuitBreaker.eventConsumerBufferSize=10

STEP-2: Create FallbackController.class
========================================

@RestController
public class FallbackController {
    @RequestMapping("/fallback/booking")
    public Mono<ResponseEntity<Map<String,Object>>> bookingFallback()
    {
        Map<String,Object> response=new HashMap<>();
        response.put("status",503);
        response.put("message","Booking Service is currently unavailable. Please try again later.");
        response.put("errorCode","BOOKING_SERVICE_DOWN");
        response.put("timestamp", Instant.now());
        return Mono.just(ResponseEntity.status(HttpStatus.SERVICE_UNAVAILABLE)
                .body(response));
    }
    @RequestMapping("/fallback/property")
    public Mono<ResponseEntity<Map<String,Object>>> propertyFallback()
    {
        Map<String,Object> response=new HashMap<>();
        response.put("status",503);
        response.put("message","Property Service is currently unavailable. Please try again later.");

                .body(response));
    }

    }

}
